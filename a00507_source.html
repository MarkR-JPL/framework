177 int AbscoHdf::wn_index(double Wn_in) const
178 {
179  if(!have_data(Wn_in)) {
180  Exception e;
181  e << "The ASBCO file " << hfile->file_name() << "\n"
182  << " does not have data for wavenumber " << Wn_in;
183  throw e;
184  }
185  double *wnptr = std::lower_bound(wnfront, wnback, Wn_in);
186  double f = (Wn_in - *(wnptr - 1)) / (*wnptr - *(wnptr - 1));
187  if(f > 0.1 && f < 0.9) {

190  << "The interpolation doesn't work well near peaks, so we\n"

192  << "of the absco wn grid, return the value for the closest number.\n"
193  << "Wn: " << Wn_in << "\n"
194  << "Wnptr - 1: " << *(wnptr - 1) << "\n"
195  << "Wnptr: " << *wnptr << "\n"
196  << "Frac: " << f << " (should be < 0.1 or > 0.9)\n"
197  << "AbscoHdf:\n"
198  << *this << "\n";
199  throw e;
200  }
201  if(f < 0.5) // First point is closest.
202  --wnptr;

233 template<class T> void AbscoHdf::swap(int i) const
234 {
235  // First time through, set up space for cache.
236  if(read_cache<T>().extent(fourthDim) == 0)
237  read_cache<T>().resize(tgrid.rows(), tgrid.cols(),
238  std::max(1, number_broadener_vmr()),
239  cache_nline);
240  int nl = read_cache<T>().extent(fourthDim);
241  // Either read 3d or 4d data. We tell which kind by whether or not
242  // we have a number_broadener_vmr() > 0 or not.
243  TinyVector<int, 4> start, size;
244  start = 0, 0, 0, (i / nl) * nl;
245  size = tgrid.rows(), tgrid.cols(), std::max(number_broadener_vmr(), 1),
246  std::min(nl, wn.rows() - start(3));
247  bound_set<T>((i / nl) * nl, size(3));
248  if(number_broadener_vmr() > 0) {
249  // 4d case
250  read_cache<T>()(Range::all(), Range::all(), Range::all(), Range(0, size(3) - 1)) =
251  hfile->read_field<T, 4>(field_name, start, size);
252  } else {
253  // 3d case
254  TinyVector<int, 3> start2, size2;
255  start2 = 0, 0, (i / nl) * nl;
256  size2 = tgrid.rows(), tgrid.cols(),
257  std::min(nl, wn.rows() - start(3));
258  read_cache<T>()(Range::all(), Range::all(), 0, Range(0, size(3) - 1)) =
259  hfile->read_field<T, 3>(field_name, start2, size2);
260  }
