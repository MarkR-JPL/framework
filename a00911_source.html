201  std::cout << "# Surface type:\n " << surface_type() << "\n"
202  << "# Surface paramters:\n" << surface_parameters << "\n"

218  //-----------------------------------------------------------------------
226  //-----------------------------------------------------------------------

233  for(int m = 0; m < jac_iv.rows(); ++m)
