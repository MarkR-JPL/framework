294 void LidortRtDriver::setup_sphericity(double zen) const
295 {
296 
297  Lidort_Fixed_Boolean& fboolean_inputs = lidort_interface_->lidort_fixin().f_bool();
298  Lidort_Modified_Boolean& mboolean_inputs = lidort_interface_->lidort_modin().mbool();
299  Lidort_Fixed_Control& fcontrol_inputs = lidort_interface_->lidort_fixin().cont();
300 
301  if(do_multi_scatt_only_) {
302  // Do not do full reflectance calculation if only multiple scattering needed
303  // These flags should already be false, but just in case..
304 
305  // Only return diffuse multiple scattering component
306  fboolean_inputs.ts_do_fullrad_mode(false);
307  fboolean_inputs.ts_do_ssfull(false);
308 
309  // Also SS corr and plane-parallel flags are false
310  mboolean_inputs.ts_do_sscorr_nadir(false);
311  mboolean_inputs.ts_do_sscorr_outgoing(false);
312 
313  } else { // if not do multi_scatt_only
314  // Do full SS + MS calculation and use LOS correction
315 
316  // Do a full reflectance calculation
317  fboolean_inputs.ts_do_fullrad_mode(true);
318  fboolean_inputs.ts_do_ssfull(false);
319 
320  // Pseudo-spherical + Line of Sight correction
321  mboolean_inputs.ts_do_sscorr_nadir(false);
322  mboolean_inputs.ts_do_sscorr_outgoing(true);
323 
324  // Number of fine layers subdividing coarse layering
325  // Only used during LOS correction
326  fcontrol_inputs.ts_nfinelayers(4);
327  }
328 
329  // Flag for controlling azimuth dependence in the output
330  // LIDORT will complain if user zenith is 0 and this is not
331  // set, when not using ss correction mode
332  if( pure_nadir_ ) {
333  mboolean_inputs.ts_do_no_azimuth(true);
334  if (mboolean_inputs.ts_do_sscorr_outgoing()) {
335  // Use Pseudo-spherical correction instead for these small viewing zenith angles
336  mboolean_inputs.ts_do_sscorr_outgoing(false);
337  mboolean_inputs.ts_do_sscorr_nadir(true);
338  }
339  }

435 void LidortRtDriver::setup_optical_inputs(const blitz::Array<double, 1>& od,
436  const blitz::Array<double, 1>& ssa,
437  const blitz::Array<double, 2>& pf) const
438 {
439 
440  // Ranges for copying inputs to method
441  Range rlay(0, od.extent(firstDim) - 1);
442  Range rmom(0, pf.extent(firstDim) - 1);
443 
444  // Convienence references
445  Lidort_Fixed_Optical& foptical_inputs = lidort_interface_->lidort_fixin().optical();
446  Lidort_Modified_Optical& moptical_inputs = lidort_interface_->lidort_modin().moptical();
447 
448  // Vertical optical depth thicness values for all layers and threads
449  Array<double, 2> deltau( foptical_inputs.ts_deltau_vert_input() );
450  deltau(rlay, 0) = od;
451 
452  // Single scattering albedos for all layers and threads
453  Array<double, 2> omega( moptical_inputs.ts_omega_total_input() );
454  omega(rlay, 0) = where(ssa > 0.999, 0.999999, ssa);

461  Array<double, 3> phasmoms( foptical_inputs.ts_phasmoms_total_input() );
462  phasmoms(rmom, rlay, 0) = where(abs(pf) > 1e-11, pf, 1e-11);

491  Lidort_Fixed_Linoptical& linoptical = lidort_interface_->lidort_linfixin().optical();
492 
493  // Number of profile weighting functions in layer n
494  int natm_jac = od.number_variable();
495  Array<int, 1> layer_jac_number( lincontrol.ts_layer_vary_number() );
496  layer_jac_number = natm_jac;
497 
498  // Ranges for copying inputs to method
499  Range ra(Range::all());
500  Range rlay(0, od.rows() - 1); // number of layers
501  Range rjac(0, natm_jac - 1);
502 
503  Range rmom(0, pf.rows() - 1); // number phase function moments
504  Range all(Range::all());
505 
506  // Flag for output of profile Jacobians
507  // Unlikely we won't need these
508  lincontrol.ts_do_profile_linearization(true);
509 
510  // Flag for output of surface Jacobians
511  // Certainly wouldn't need this if not retrieving ground
512  lincontrol.ts_do_surface_linearization(do_surface_linearization);
513 
514  // Check that we fit within the LIDORT configuration
515  if(linoptical.ts_l_deltau_vert_input().cols() < od.rows()) {

538  throw e;
539  }
540 
541  // Setup optical linear inputs
542  // LIDORT expects the following:
543  // l_deltau = xi/tau * dtau/dxi
544  // l_omega = xi/omega * domega/dxi
545  // ... etc
546  // The driver is handed dtau/dxi, domega/dxi ... etc
547  // Where it is normally set up for xi being taug, taur, taua[0], taua[1]...
548  //
549  // LIDORT returns to us:
550  // xi * dI/dxi
551  // Therefore you will notice that by not multiplying dtau/dxi by xi and only
552  // dividing by xi, we are cancelling out the xi in the result and hence
553  // the driver really return dI/dxi
554  Array<double, 2> l_deltau( linoptical.ts_l_deltau_vert_input()(rjac,rlay,0) );
555  Array<double, 2> l_omega( linoptical.ts_l_omega_total_input()(rjac,rlay,0) );
556  Array<double, 3> l_phasmoms( linoptical.ts_l_phasmoms_total_input()(rjac,rmom,rlay,0) );
557 
558  // Transpose these to match dimensions used internally
559  l_deltau.transposeSelf(secondDim, firstDim);
560  l_omega.transposeSelf(secondDim, firstDim);
561 
562  // Note that LIDORT does *not* take l_deltau etc. Rather it takes
563  // what it calls "normalized derivative inputs". So for an optical
564  // quantity like taug wrt to tau, this would be taug / tau *
565  // dtau/dtaug.
566  //
567  // It then returns a normalized jacobian, which for taug would be
568  // taug * d I /dtaug.
569  //
570  // Note that we actually leave out one of these factors, because it
571  // effectively cancels out. We pass in 1 / tau * dtau / dtaug and
572  // get back d I / dtaug.
573 
574  firstIndex i1; secondIndex i2;
575  l_deltau = where(od.value()(i1) != 0, od.jacobian() / od.value()(i1), 0.0);
576 
577  Array<double, 1> ssa_limit(ssa.rows());
578  ssa_limit = where(ssa.value() > 0.999, 0.999999, ssa.value());
579  l_omega = where(ssa_limit(i1) != 0, ssa.jacobian() / ssa_limit(i1), 0.0);
580 
581  if(pf.is_constant())
582  l_phasmoms(rjac, rmom, rlay) = 0.0;
583  else {
584  blitz::Array<double, 2> pf_in( where(abs(pf.value()) > 1e-11, pf.value(), 1e-11) );
585 
586  // We need this loop since l_phasmoms and pf variables have jacobian data in different dimensions
587  for (int jidx = 0; jidx < pf.number_variable(); jidx++)
588  l_phasmoms(jidx, rmom, rlay) = pf.jacobian()(rmom, rlay, jidx) / pf_in(rmom, rlay)(i1,i2);
589  }
590 }
