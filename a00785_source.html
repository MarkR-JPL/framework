284 void TwostreamRtDriver::setup_optical_inputs(const blitz::Array<double, 1>& od,
285  const blitz::Array<double, 1>& ssa,
286  const blitz::Array<double, 2>& pf) const
287 {
288  // Ranges for copying inputs to method
289  Range rlay(0, od.extent(firstDim) - 1);
290 
291  // Vertical optical depth thicness values for all layers and threads
292  Array<double, 2> deltau( twostream_interface_->deltau_input() );
293  deltau(rlay, 0) = od;
294 
295  // Single scattering albedos for all layers and threads
296  Array<double, 2> omega( twostream_interface_->omega_input() );
297  omega(rlay, 0) = where(ssa > 0.999, 0.999999, ssa);
298 
299  // Assymetry factor
300  // Equal to one third of the first phase function moment
301  Array<double, 2> asymm_input( twostream_interface_->asymm_input() );
302  asymm_input(rlay, 0) = pf(1, rlay) / 3.0;
303 
304  // Delta-m scaling factor for 2-stream
305  // Equal to one fifth of the second phase function moment
306  if (twostream_interface_->do_d2s_scaling()) {
307  Array<double, 2> d2s_scaling( twostream_interface_->d2s_scaling() );
308  d2s_scaling(rlay, 0) = pf(2, rlay) / 5.0;
309  }

319 void TwostreamRtDriver::setup_linear_inputs(const ArrayAd<double, 1>& od,
320  const ArrayAd<double, 1>& ssa,
321  const ArrayAd<double, 2>& pf,
322  bool do_surface_linearization) const
323 {
324  // Number of profile weighting functions in layer n

326  Array<int, 1> layer_jac_number( twostream_interface_->layer_vary_number() );
327  layer_jac_number = natm_jac;
328 
329  // Ranges for copying inputs to method
330  Range rlay(0, od.rows() - 1); // number of layers
331  Range rjac(0, natm_jac - 1);
332  Range all(Range::all());
333 
334  // Flag for output of profile Jacobians
335  // Unlikely we won't need these
336  twostream_interface_->do_profile_wfs(true);
337 
338  // Needs to match the number set in the BRDF structure
339  twostream_interface_->n_surface_wfs(brdf_driver_->n_surface_wfs());
340 
341  // Flag for output of surface Jacobians
342  // Certainly wouldn't need this if not retrieving ground
343  twostream_interface_->do_surface_wfs(do_surface_linearization);
344 
345  // Check that we fit within the LIDORT configuration
346  if(twostream_interface_->l_deltau_input().rows() < od.rows()) {
347  Exception e;
348  e << "The number of layers you are using exceeds the maximum allowed by\n"
349  << "the current build of Lidort. The number requested is "
350  << od.rows() << "\nand the maximum allowed is "
351  << twostream_interface_->l_deltau_input().rows() << "\n"
352  << "\n"
353  << "You might try rebuilding with a larger value given to the configure\n"
354  << "option --with-lidort-maxlayer=value set to a larger value.\n";
355  throw e;
356  }
357  if(twostream_interface_->l_deltau_input().cols() < natm_jac) {
358  Exception e;
359  e << "The number of jacobians you are using exceeds the maximum allowed by\n"
360  << "the current build of Lidort. The number requested is "
361  << natm_jac << "\nand the maximum allowed is "
362  << twostream_interface_->l_deltau_input().cols() << "\n"
363  << "\n"
364  << "This number of jacobians is a function of the number of aerosols\n"
365  << "in your state vector, so you can reduce the number of aerosols\n"
366  << "\n"
367  << "You might also try rebuilding with a larger value given to the configure\n"
368  << "option --with-lidort-maxatmoswfs=value set to a larger value.\n";
369  throw e;
370  }
371 
372  // Setup optical linear inputs
373  Array<double, 2> l_deltau( twostream_interface_->l_deltau_input()(rlay,rjac,0) );
374  Array<double, 2> l_omega( twostream_interface_->l_omega_input()(rlay,rjac,0) );
375  Array<double, 2> l_asymm_input( twostream_interface_->l_asymm_input()(rlay,rjac,0) );
376  Array<double, 2> l_d2s_scaling( twostream_interface_->l_d2s_scaling()(rlay,rjac,0) );
377 
378  l_deltau = od.jacobian();
379  l_omega = ssa.jacobian();
380 
381  l_asymm_input = pf.jacobian()(1, all, all) / 3.0;
382  l_d2s_scaling = pf.jacobian()(2, all, all) / 5.0;
