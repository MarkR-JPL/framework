205 // Azimuth is modified because the convention used by the OCO L1B file is to
206 // take both solar and observation angles as viewed from an observer
207 // standing in the FOV. In this convention, the convention for glint
208 // would be a relative azimuth difference of 180 degrees, as the
209 // spacecraft and sun would be on opposite sides of the sky. However, the
210 // radiative transfer convention is that the azimuth angles must be the
211 // same for glint (it is "follow the photons" convention). However, we'd
212 // like the solar azimuth to not be changed, so as to continue to agree
213 // with zenith, so this change of the observation azimuth has the effect
214 // of putting everything in a "reverse follow-the-photons" convention,
215 // where we look from the satellite to the FOV, then from the FOV to the
216 // sun. Note that because of an old historical reason, however, both
217 // zenith angles remain > 0 and < 90, even in the RT convention.
218 
219 DoubleWithUnit Level1b::relative_azimuth(int Spec_index) const
220 {
221  Unit orig_unit = sounding_azimuth(Spec_index).units;
222  double res = (180 + sounding_azimuth(Spec_index).convert(units::deg).value) -
223  solar_azimuth(Spec_index).convert(units::deg).value;
224 
225  if (res > 360)
226  res -= 360;
227  else if(res < 0)
228  res += 360;
229 
230  return DoubleWithUnit(res, units::deg).convert(orig_unit);
231 }
